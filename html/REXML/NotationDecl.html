def to_s
  c = nil
  c = parent.context if parent
  if c and c[:prologue_quote] == :apostrophe
    quote = "'"
  else
    quote = "\""
  end
  notation = "<!NOTATION #{@name} #{@middle}"
  notation << " #{quote}#{@public}#{quote}" if @public
  notation << " #{quote}#{@system}#{quote}" if @system
  notation << ">"
  notation
end
